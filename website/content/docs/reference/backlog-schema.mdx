# Backlog JSON Schema

This is the formal schema for `backlog.json` files used by `qap project import`.

## JSON Schema

```json
{
  "$schema": "https://json-schema.org/draft/2020-12/schema",
  "title": "BacklogManifest",
  "description": "Backlog manifest for qap project import",
  "type": "object",
  "required": ["version", "project", "epics", "tasks"],
  "properties": {
    "version": {
      "const": 1,
      "description": "Schema version. Must be 1."
    },
    "project": {
      "type": "object",
      "required": ["id", "name"],
      "properties": {
        "id": {
          "type": "string",
          "description": "Project identifier (slug)"
        },
        "name": {
          "type": "string",
          "description": "Human-readable project name"
        },
        "tracker": {
          "type": "object",
          "required": ["provider"],
          "properties": {
            "provider": {
              "enum": ["linear"],
              "description": "Issue tracker provider"
            },
            "projectId": {
              "type": "string",
              "description": "Tracker project identifier"
            },
            "teamId": {
              "type": "string",
              "description": "Tracker team identifier"
            }
          }
        }
      }
    },
    "sharedContext": {
      "type": "string",
      "description": "Filename of shared context file relative to prompts directory"
    },
    "epics": {
      "type": "array",
      "minItems": 1,
      "items": {
        "type": "object",
        "required": ["id", "title", "track"],
        "properties": {
          "id": {
            "type": "string",
            "description": "Unique epic identifier"
          },
          "title": {
            "type": "string",
            "description": "Epic title"
          },
          "track": {
            "enum": ["main", "sidecar", "gate"],
            "description": "Execution priority track"
          },
          "promptFile": {
            "type": "string",
            "description": "Epic prompt file relative to prompts directory"
          }
        }
      }
    },
    "tasks": {
      "type": "array",
      "minItems": 1,
      "items": {
        "type": "object",
        "required": ["id", "title", "epicId"],
        "properties": {
          "id": {
            "type": "string",
            "description": "Unique task identifier (e.g., QUE-238)"
          },
          "title": {
            "type": "string",
            "description": "Task title"
          },
          "epicId": {
            "type": "string",
            "description": "Reference to parent epic ID"
          },
          "kind": {
            "enum": ["implementation", "validation", "cleanup", "migration", "poc"],
            "default": "implementation",
            "description": "Task type"
          },
          "track": {
            "enum": ["main", "sidecar", "gate"],
            "default": "main",
            "description": "Execution priority track"
          },
          "dependsOn": {
            "type": "array",
            "items": { "type": "string" },
            "description": "Task IDs that must complete before this task"
          },
          "promptFile": {
            "type": "string",
            "description": "Task prompt file relative to prompts directory"
          },
          "issueUrl": {
            "type": "string",
            "format": "uri",
            "description": "URL to the issue in your tracker"
          },
          "acceptanceCriteria": {
            "type": "array",
            "items": { "type": "string" },
            "description": "Acceptance criteria displayed in the task prompt"
          }
        }
      }
    }
  }
}
```

## Validation Rules (Beyond Schema)

These rules are enforced at runtime and cannot be expressed in JSON Schema alone:

1. **Epic reference integrity** — every `task.epicId` must match an `epic.id` in the `epics` array
2. **Dependency reference integrity** — every entry in `task.dependsOn` must match a `task.id` in the `tasks` array
3. **No duplicate task IDs** — each `task.id` must be unique
4. **No dependency cycles** — the dependency graph must be a DAG (directed acyclic graph), verified via DFS

## TypeScript Type

The equivalent TypeScript interface (from `src/config/backlog.ts`):

```typescript
interface BacklogManifest {
  version: 1;
  project: {
    id: string;
    name: string;
    tracker?: {
      provider: "linear";
      projectId?: string;
      teamId?: string;
    };
  };
  sharedContext?: string;
  epics: BacklogEpic[];
  tasks: BacklogTask[];
}

interface BacklogEpic {
  id: string;
  title: string;
  track: "main" | "sidecar" | "gate";
  promptFile?: string;
}

interface BacklogTask {
  id: string;
  title: string;
  epicId: string;
  kind?: "implementation" | "validation" | "cleanup" | "migration" | "poc";
  track?: "main" | "sidecar" | "gate";
  dependsOn?: string[];
  promptFile?: string;
  issueUrl?: string;
  acceptanceCriteria?: string[];
}
```
